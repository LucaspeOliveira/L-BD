CREATE VIEW Lista_P200 AS
SELECT a.ra AS Num_Documento, 
	a.nome AS Nome_Pessoa,
	pa.titulo AS Titulo_Palestra,
	pe.nome AS Nome_Palestrante,
	pa.carga_horaria AS Carga_Horária,
	pa.data AS Data
FROM aluno AS a
INNER JOIN alunos_inscritos AS ai	ON ai.ra = a.ra
INNER JOIN palestra			AS pa	ON pa.codigo_palestra = ai.codigo_palestra
INNER JOIN palestrante		AS pe	ON pe.codigo_palestrante = pa.codigo_palestrante
WHERE pa.codigo_palestra = 200
UNION
SELECT na.rg + '-' + na.orgao_exp AS Num_Documento,
	na.nome AS Nome_Pessoa,
	pa.titulo AS Titulo_Palestra,
	pe.nome AS Nome_Palestrante,
	pa.carga_horaria AS Carga_Horária,
	pa.data AS Data
FROM nao_aluno AS na
INNER JOIN nao_alunos_inscritos AS nai	ON nai.rg = na.rg AND nai.orgao_exp = na.orgao_exp
INNER JOIN palestra				AS pa	ON pa.codigo_palestra = nai.codigo_palestra
INNER JOIN palestrante			AS pe	ON pe.codigo_palestrante = pa.codigo_palestrante
WHERE pa.codigo_palestra = 200
